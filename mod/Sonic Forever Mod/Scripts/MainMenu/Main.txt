function BacktoMain
	if inputPress.ButtonB == 1
		menu.state = prev.state
		prev.state = 0
	end if
end function

function AchieveLock

end function

function GameplayUpDown
	if inputPress.up == 1
		menu.state--
		if menu.state < 10
			menu.state = 27
			menu.state += saveRAM[417]
		end if
	end if
	
	if inputPress.down == 1
		menu.state++
		temp0 = 27
		temp0 += saveRAM[417]
		if menu.state > temp0
			menu.state = 10
		end if
	end if
end function

function BacktoSega
	if inputPress.ButtonB == 1
		menu.state = 8
		prev.state = 0
	end if
end function

function BacktoExtras
	if inputPress.ButtonB == 1
		menu.state = 100
	end if
end function
